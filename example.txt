    <script>
        // 1. Navigation Logic
        function navigate(targetId) {
            // Hide all sections
            document.querySelectorAll('.page-section').forEach(section => {
                section.classList.remove('active');
            });
            
            // Remove active state from all nav links
            document.querySelectorAll('.nav-link').forEach(link => {
                link.classList.remove('active');
            });

            // Show target section
            document.getElementById(targetId).classList.add('active');
            
            // Highlight current nav button (simple matching based on text content)
            const btnMap = { 'landing': 'Home', 'builder': 'Card Builder', 'login': 'Log In' };
            document.querySelectorAll('.nav-link').forEach(link => {
                if(link.textContent === btnMap[targetId]) {
                    link.classList.add('active');
                }
            });
        }

        // 2. Real-Time Data Binding for the Builder
        const dataMapping = {
            'inputName': { target: 'previewName', default: 'Your Name' },
            'inputTitle': { target: 'previewTitle', default: 'Your Title' },
            'inputCompany': { target: 'previewCompany', default: 'Your Company' },
            'inputBio': { target: 'previewBio', default: 'Your professional biography will appear here.' },
            'inputEmail': { target: 'previewEmail', default: 'Email: -', prefix: 'Email: ' },
            'inputPhone': { target: 'previewPhone', default: 'Phone: -', prefix: 'Phone: ' },
            'inputWebsite': { target: 'previewWebsite', default: 'Web: -', prefix: 'Web: ' }
        };

        Object.keys(dataMapping).forEach(inputId => {
            const inputElement = document.getElementById(inputId);
            const previewElement = document.getElementById(dataMapping[inputId].target);
            const defaultValue = dataMapping[inputId].default;
            const prefix = dataMapping[inputId].prefix || '';

            inputElement.addEventListener('input', function(e) {
                const value = e.target.value.trim();
                if (value === '') {
                    previewElement.textContent = defaultValue;
                } else {
                    previewElement.textContent = prefix + value;
                }
            });
        });
    </script>
